//Stack import java.util.Scanner; public class Stack { public static void main(String[] args) { final int MAX_SIZE = 10; int[] stack = new int[MAX_SIZE]; int top = -1; Scanner scanner = new Scanner(System.in); while (true) { System.out.println("1. Push"); System.out.println("2. Pop"); System.out.println("3. Display"); System.out.println("4. Exit"); System.out.print("Enter your choice: "); int choice = scanner.nextInt(); if (choice == 1) { if (top == MAX_SIZE - 1) { System.out.println("Stack is full!"); } else { System.out.print("Enter the element to push: "); int element = scanner.nextInt(); top = push(stack, top, element); System.out.println("Element pushed to stack: " + element); } } else if (choice == 2) { if (top == -1) { System.out.println("Stack is empty!"); } else { int[] result = pop(stack, top); top = result[0]; System.out.println("Popped element: " + result[1]); } } else if (choice == 3) { display(stack, top); } else if (choice == 4) { break; } else { System.out.println("Invalid choice! Please enter a valid option."); } } scanner.close(); } public static int push(int[] stack, int top, int element) { top++; stack[top] = element; return top; }
